    public IList GetAllByParent(int universityId)
    {
        var result = DBHelper.GetData(
        new EducationalProgramResultDTOMapper(), 
        $"SELECT [UniversityId], " +
        $"      [Univer_Go].[dbo].[Speciality].[SpecialityCode], " +
        $"      [Univer_Go].[dbo].[Speciality].[Name], " +
        $"      [Univer_Go].[dbo].[StudyCondition].[Price], " +
        $"      [Univer_Go].[dbo].[StudyCondition].[Id], " +
        $"      [Univer_Go].[dbo].[Profiles].[profile] " +
        $"FROM [Univer_Go].[dbo].[UniversitySpeciality] " +
        $"  join " +
        $"      [Univer_Go].[dbo].[Speciality] " +
        $"  on " +
        $"      [SpecialityId] = [Univer_Go].[dbo].[Speciality].[Id]" +
        $"  join " +
        $"      [Univer_Go].[dbo].[StudyCondition] " +
        $"  on " +
        $"      [Univer_Go].[dbo].[StudyCondition].[UniversitySpecialityId] = [Univer_Go].[dbo].[UniversitySpeciality].[Id]  " +
        $"  join" +
        $"      [Univer_Go].[dbo].[Profiles] " +
        $"  on" +
        $"      [Univer_Go].[dbo].[Profiles].[id] = [Univer_Go].[dbo].[StudyCondition].[ProfileId] " +
        $"WHERE UniversityId='{universityId}'");
             
        return result;
    }
